Uso Básico:

from app.ssrf_scanner import scan_url_for_ssrf

findings = scan_url_for_ssrf(
    url="https://target.com/api?url=https://target.com",
    callback_host="your-callback.com",  # Opcional para blind SSRF
    headers={"Authorization": "Bearer token"}  # Opcional
)

Verificar Resultados:

for finding in findings:
    print(f"Severity: {finding['severity']}")           # critical/high
    print(f"Type: {finding['ssrf_type']}")              # cloud_metadata/local_file_read/internal_access
    print(f"Parameter: {finding['param_name']}")        # url, uri, target, etc
    print(f"Payload: {finding['payload']}")             # Payload utilizado
    print(f"Source: {finding.get('source', 'native')}")  # bugbounty_cheatsheet ou native_scanner
